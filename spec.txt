<unnamable-rule> ::= <universe-def> | <sonority-def> | <language-allowance>
	<universe-def> ::= <name> <universe-op> ( <name> ) 
		<universe-op> ::= <quantifier-existential> | <quantifier-universal>
	<sonority-def> ::= ( <sonority-def> <seperator-sonority> ) <name> <seperator-sonority> <name>
	<language-allowance> ::= <seperator-allow> { <feature-vec> | <group> | <name> }

<name-assign> ::= <name> <seperator-name> { <nameable-rule> | <group> }
	<prob-assign> ::= <seperator-probability> <probability>

	<nameable-rule> ::= <sound-change> | <feature-vec> | <environment> | <feat-vec-str>
		<feature-vec> ::= { <feature-vec-def> | <name> | <group> } ( <prob-assign> )
			<feature-vec-def> ::= <begin-feature-vector> <feature-str> <end-feature-vector>
				<feature-str> ::= ( <feature-str> <seperator-list> ) <spec>
					<spec> ::= <spec-op> <name>
						<spec-op> ::= <feature-positive> | <feature-negative>

		<feat-vec-str> ::= { <feat-vec-str-def> | <name> | <group> } ( <prob-assign> )
			<feat-vec-str-def> ::= ( <feat-vec-str> ) <feature-vec> <feature-vec>

		<environment> ::= { <environment-def> | <name> | <group> } ( <prob-assign> )
			<environment-def> ::= { <feat-vec-str> | <feature-vec> } <seperator-environment> { <feat-vec-str> | <feature-vec> }

		<sound-change> ::= { <sound-change-def> | <name> | <group> } ( <prob-assign> )
			<sound-change-def> ::= <feature-vec> <seperator-sound-change> <feature-vec> <seperator-sound-change> <environment>
	
	<group> ::= <group-def> | <name>
		<group-def> ::= <begin-group> { <nameable-rule-str> | <query-str> | <op-cat-str> } <end-group>
			<query-str> ::= ( <query-str> <seperator-list> ) <query>
			<nameable-rule-str> ::= ( <nameable-rule-str> <seperator-list> ) <nameable-rule>
			<op-cat-str> ::= ( <op-cat-str> <seperator-list> ) <op-cat>

<op-cat> ::= ( <op-cat> <seperator-concatenate> ) <op-cat-arg> <seperator-concatenate> <op-cat-arg>
	<op-cat-arg> ::= <name> | <query> | <group> | <feat-vec-str> | <feature-vec>

<query> ::= <generate> | <apply>
	<target> ::= <feature-vec> | <feat-vec-str> | <group> | <name>
	
	<generate> ::= <number> <seperator-generate> <target>
	<apply> ::= <sound-change> <seperator-apply> { <target> | <op-cat> }

<program> ::= ( <program> ) <statement> <seperator-statement>
	<statement> ::= <unnamable-rule> | <name-assign> | <query>
